\[

\[

\[
f\left( x\right)  = \mathop{\sum }\limits_{{n = 1}}^{\infty }\frac{{\left( -1\right) }^{n}}{{2n} + 1}{x}^{{2n} + 1}, x \in  \left\lbrack  {-1,1}\right\rbrack  .

\[
{f}^{\prime }\left( x\right)  = \mathop{\sum }\limits_{{n = 1}}^{\infty }{\left( -1\right) }^{n}{x}^{2n} =  - \frac{{x}^{2}}{1 + {x}^{2}} = \frac{1}{1 + {x}^{2}} - 1, x \in  \left( {-1,1}\right) .

\[
